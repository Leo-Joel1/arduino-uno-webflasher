    const log = (msg) => {
      document.getElementById("log").textContent += msg + "\n";
    };

    async function flashUno() {
      try {
        log("Connecting to Arduino Cloud Agent...");
        const client = new ArduinoCreateAgentClient();

        await client.init(); // handshake with Agent
        log("Agent connected!");

        // List available ports
        const ports = await client.listPorts();
        if (!ports.length) {
          log("No boards detected!");
          return;
        }
        log("Board detected: " + ports[0].address);

        // Fetch firmware
        const firmwareUrl = "Digital_Clock-Timer_aug20a.ino.hex";
        log("Fetching firmware...");
        const hex = await fetch(firmwareUrl).then(r => r.text());

        // Upload firmware
        log("Uploading...");
        await client.flash(ports[0].address, hex);
        log("Upload complete!");
      } catch (err) {
        log("Error: " + err);
      }
    }

    document.getElementById("flash").addEventListener("click", flashUno);
